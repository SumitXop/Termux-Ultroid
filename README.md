# install Ultroid

```

bash ubuntu-start.sh

apt update && apt upgrade
apt install --fix-broken
apt install git
git clone https://github.com/rooted-cyber/Termux-Ultroid
cd Termux-Ultroid
bash install-ultroid.sh
bash start-ultroid.sh
```
